# 安装程序
```
echo -ne "程序正在安装"
apt-get install PackageName -y >/dev/null 2>&1 &
PID=$!
echo -ne " ... "
while kill -0 $PID 2>/dev/null
do
    SPINLINE[0]="-"
    SPINLINE[1]="\\"
    SPINLINE[2]="|"
    SPINLINE[3]="/"
    for SPIN in "${SPINLINE[@]}"
    do
        echo -ne "$SPIN"
        sleep 0.1
        echo -ne "\b"
    done
done
if [ $? -eq "0" ]; then
    echo "PASS"
fi

#显示效果：
#安装时：程序正在安装 ... /
#安装完成：程序正在安装 ... PASS
```
