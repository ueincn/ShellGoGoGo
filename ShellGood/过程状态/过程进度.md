# 进度条
```
function ProgressBar(){
    BARLINE="+"
    for ((NUM=0;NUM<=100;NUM+=5))
    do
        printf "进度条：[%-40s]\r" "$BARLINE"
        BARLINE=++$BARLINE
        sleep 0.2
    done
}

#打印状态：
#进度条：[+++++++++++++++++++++++++++++++++++++++++]
```

# 进度值
```
function ProgressValue(){
    for ((NUM=0;NUM<=100;NUM+=5))
    do
        printf "进度数值：%d%%\r" "$NUM"
        sleep 0.2
    done
}

#打印状态：
#进度数值：100%
```
